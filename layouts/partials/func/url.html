{{- /* url — internal link helper, base-path aware. Prepends baseURL's sub-path to internal paths; external URLs (http/mailto/tel) and #anchors pass through, so it's safe to wrap every href. Uses baseURL's path rather than relURL, which drops the sub-path on leading-slash paths. "/projects" → /projects (or /sub/projects) "#top" → #top */ -}} {{- $p := . -}} {{- $out := $p -}} {{- if not $p -}} {{- $out = (strings.TrimSuffix "/" (urls.Parse site.BaseURL).Path) | default "/" -}} {{- else if or (hasPrefix $p "http") (hasPrefix $p "mailto:") (hasPrefix $p "tel:") (hasPrefix $p "#") -}} {{- $out = $p -}} {{- else -}} {{- if not (hasPrefix $p "/") }}{{ $p = printf "/%s" $p }}{{ end -}} {{- $base := strings.TrimSuffix "/" (urls.Parse site.BaseURL).Path -}} {{- $out = printf "%s%s" $base $p -}} {{- end -}} {{- return $out -}}
